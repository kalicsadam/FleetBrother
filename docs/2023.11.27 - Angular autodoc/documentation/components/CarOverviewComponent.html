import { Component, EventEmitter, Input, Output } from '@angular/core';
import { Car } from 'src/app/data/dto/car.dto';

export enum CarOverviewMode {
  ACCEPT_DECLINE,
  MANAGE,
  REPORT,
  DISPLAY
}

@Component({
  selector: 'app-car-overview',
  templateUrl: './car-overview.component.html',
  styleUrls: ['./car-overview.component.scss']
})
export class CarOverviewComponent {
  modetype = CarOverviewMode;
  @Input() mode : CarOverviewMode = CarOverviewMode.ACCEPT_DECLINE;

  @Input() car : Car | undefined;
  @Input() disabled : boolean = false;
  @Output() onAccept : EventEmitter<Car> = new EventEmitter();
  @Output() onDecline : EventEmitter<Car> = new EventEmitter();
  @Output() onDelete : EventEmitter<Car> = new EventEmitter();
  @Output() onAlerts : EventEmitter<Car> = new EventEmitter();
}

<mat-card *ngIf="car != null" class="car-overview">
    <mat-card-header>
        <mat-card-title>
            <p>#{{car.id}}: {{car.name}}</p>
        </mat-card-title>
        <mat-card-subtitle>
            <p>License plate: {{car.licensePlate}}</p>
            <p>VIN: {{car.vin}}</p>
        </mat-card-subtitle>
    </mat-card-header>
    <mat-card-content class="car-overview-content">
        <div class="car-overview-tools">
            <div *ngIf="mode == modetype.ACCEPT_DECLINE" class="tools-accept-decline">
                <button mat-fab color="accent" (click)="onAccept.emit(car)" [disabled]="disabled">
                    <mat-icon>done</mat-icon>
                  </button>
                <button mat-fab color="warn" (click)="onDecline.emit(car)" [disabled]="disabled">
                    <mat-icon>close</mat-icon>
                </button>
            </div>
            <div *ngIf="mode == modetype.MANAGE" class="tools-report">
                <button mat-fab color="primary" [routerLink]="['/report', car.id]" [disabled]="disabled">

                </button>
                <button mat-fab color="warn" (click)="onDelete.emit(car)" [disabled]="disabled">
                    <mat-icon>delete</mat-icon>
                </button>
            </div>
            <div *ngIf="mode == modetype.REPORT">
                <button mat-fab color="warn" (click)="onAlerts.emit(car)" [disabled]="disabled">

    </mat-card-content>
</mat-card>
