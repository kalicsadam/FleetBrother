import { Component, EventEmitter, Input, Output } from '@angular/core';
import { User } from 'src/app/data/dto/user.dto';

@Component({
  selector: 'app-user-overview',
  templateUrl: './user-overview.component.html',
  styleUrls: ['./user-overview.component.scss']
})
export class UserOverviewComponent {
  @Input() user : User | undefined

  @Output() onDelete: EventEmitter<User> = new EventEmitter()
}

<mat-card *ngIf="user != null">
    <mat-card-content class="user-details">
        <div>
            <mat-card-title>{{user.email}}</mat-card-title>
            <mat-card-subtitle>Is Admin: {{user.isAdmin}}</mat-card-subtitle>
        </div>
        <button mat-icon-button (click)="onDelete.emit(user)">
            <mat-icon>delete</mat-icon>
        </button>
    </mat-card-content>
</mat-card>
